// Twinkling foil particles on owned cards

function drawFoilSparkles(ctx, slot, cardId, slotWidth, slotHeight) {

    ctx.save();

    const sparkleSeed = cardId * 999;

    for (let i = 0; i < 3; i++) {

        const px = slot.x + 30 + ((sparkleSeed * (i + 1) * 17) % (slotWidth - 60));

        const phase = (Date.now() / 250 + i * 2.2) % (Math.PI * 2);

        const size = (Math.sin(phase) + 1) * 2.5 + 0.5;


        if (size > 1.2) {

            ctx.beginPath();

            ctx.moveTo(px, py - size * 2);

            ctx.quadraticCurveTo(px, py, px + size * 2, py);

            ctx.quadraticCurveTo(px, py, px, py + size * 2);

            ctx.quadraticCurveTo(px, py, px - size * 2, py);

            ctx.quadraticCurveTo(px, py, px, py - size * 2);

            ctx.fill();

        }

    }

    ctx.restore();

        const py = slot.y + 25 + ((seed * (i + 3) * 59) % (slotHeight - 50));


        const wave = (Date.now() / 180 + i * 1.5) % (Math.PI * 2);

        const radius = (Math.sin(wave) + 1) * 3 + 1;


        const hue = (Date.now() / 20 + i * 90) % 360;

        ctx.fillStyle = `hsla(${hue}, 100%, 85%, 0.95)`;


        ctx.beginPath();

        ctx.arc(px, py, radius / 2, 0, Math.PI*2);

        ctx.fill();


        ctx.strokeStyle = `hsla(${hue}, 100%, 75%, 0.45)`;

        ctx.lineWidth = 1;

        ctx.beginPath();

        ctx.moveTo(px - radius * 2, py);

        ctx.lineTo(px + radius * 2, py);

        ctx.moveTo(px, py - radius * 2);

        ctx.lineTo(px, py + radius * 2);

        ctx.stroke();

    }

    ctx.restore();

}

        let endY = 0;


        if (e.changedTouches && e.changedTouches.length > 0) {

            endY = e.changedTouches[0].clientY;

            return;

// Scryfall Database dynamic Search Query

async function handleGrailSearch() {

    const input = document.getElementById('grail-search-input');

    const query = input.value.trim();

    if (!query) return;


    const btn = document.getElementById('btn-grail-search');

    const select = document.getElementById('grail-print-select');

    const addBtn = document.getElementById('btn-grail-add');

    const statusMsg = document.getElementById('grail-search-status');


    btn.disabled = true;

    statusMsg.innerText = "⚡ Invoking Scryfall database...";

    statusMsg.className = "text-xs text-purple-400 font-medium";

    select.innerHTML = '<option value="">-- Printing Search Running --</option>';

    select.disabled = true;

    addBtn.disabled = true;


    try {

        const url = `https://api.scryfall.com/cards/search?q=${encodeURIComponent(query)}&unique=prints`;

        const response = await fetch(url);


        if (!response.ok) {

            throw new Error("No printings found.");

        }


        const data = await response.json();

        if (!data.data || data.data.length === 0) {

            throw new Error("No cards resolved.");

        }


        currentSearchCards = data.data;

        const prints = data.data.slice(0, 100);


        select.innerHTML = '';

        prints.forEach((card, index) => {

            const opt = document.createElement('option');

            opt.value = index;

            const setName = card.set_name;

            const setCode = card.set.toUpperCase();

            const num = card.collector_number;

            const rarity = card.rarity.charAt(0).toUpperCase() + card.rarity.slice(1);

            opt.innerText = `${card.name} | ${setName} (${setCode}) #${num} [${rarity}]`;

            select.appendChild(opt);

        });


        select.disabled = false;

        addBtn.disabled = false;

        statusMsg.innerText = `🔮 Summoned ${data.total_cards || data.data.length} variants of ${prints[0].name}!`;

        statusMsg.className = "text-xs text-emerald-400 font-medium animate-pulse";

    } catch (err) {

        statusMsg.innerText = "❌ Scryfall scroll failed. Card name not found.";

        statusMsg.className = "text-xs text-rose-400 font-medium";

        select.innerHTML = '<option value="">-- No printings loaded --</option>';

        select.disabled = true;

        addBtn.disabled = true;

    } finally {

        btn.disabled = false;

    }

}


function handleAddGrail() {

    const select = document.getElementById('grail-print-select');

    const selectedIndex = select.value;

    if (selectedIndex === "") return;


    const cardData = currentSearchCards[selectedIndex];

    if (!cardData) return;


    const exists = grailsCollection.some(c => c.set === cardData.set && c.num === cardData.collector_number);

    if (exists) {

        const statusMsg = document.getElementById('grail-search-status');

        statusMsg.innerText = "⚠️ That specific printing exists already inside your Grails!";

        statusMsg.className = "text-xs text-yellow-400 font-medium";

        return;

    }


    const newGrailCard = {

        id: Date.now() + Math.floor(Math.random() * 1000), 

        name: cardData.name,

        set: cardData.set,

        num: cardData.collector_number,

        rarity: cardData.rarity.charAt(0).toUpperCase() + cardData.rarity.slice(1)

    };


    grailsCollection.push(newGrailCard);

    collection[`grails-${newGrailCard.set}-${newGrailCard.num}`] = true;


    document.getElementById('grail-search-input').value = '';

    select.innerHTML = '<option value="">-- No printings loaded --</option>';

    select.disabled = true;

    document.getElementById('btn-grail-add').disabled = true;


    const statusMsg = document.getElementById('grail-search-status');

    statusMsg.innerText = `✅ Loaded ${cardData.name} into your custom Vault Grails page!`;

    statusMsg.className = "text-xs text-emerald-400 font-bold";


    updateStats();

    updatePaginationUI();

}

function resizeCanvas() {

    const parent = canvas.parentElement;

    const containerWidth = parent.clientWidth;

    const dpr = window.devicePixelRatio || 1;


    // Responsive css dimensions bound to canvas elements

    canvas.style.width = containerWidth + 'px';

    canvas.style.height = (containerWidth * (V_HEIGHT / V_WIDTH)) + 'px';

    canvas.height = V_HEIGHT * dpr;

    ctx.scale(dpr, dpr);

}
