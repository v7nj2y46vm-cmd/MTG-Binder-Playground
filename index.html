        function playPageTurnSound() {

        function getFilteredCards() {

            const list = activeSection === 'poster' ? POSTER_CARDS : REALMS_CARDS;

            return list.filter(card => {

                // Name match

                const matchSearch = card.name.toLowerCase().includes(searchFilter.toLowerCase());

                

                // Ownership match

                const key = `${activeSection}-${card.id}`;

                const isOwned = !!collection[key];

                const matchStatus = statusFilter === 'all' || 

                                    (statusFilter === 'owned' && isOwned) || 

                                    (statusFilter === 'missing' && !isOwned);


                return matchSearch && matchStatus;

            });

        }

                ctx.globalAlpha = 0.38;

                surgeSheen.addColorStop(1, 'rgba(255, 255, 0, 0.04)');

                ctx.fillStyle = surgeSheen;

        // Twinkling foil sparkles (glints) for owned premium cards

        function drawFoilSparkles(ctx, slot, cardId) {

            ctx.save();

            ctx.globalCompositeOperation = 'screen';

            ctx.fillStyle = '#ffffff';

            

            const sparkleSeed = cardId * 999;

            for (let i = 0; i < 3; i++) {

                const px = slot.x + 30 + ((sparkleSeed * (i + 1) * 17) % (CARD_WIDTH - 60));

                const py = slot.y + 30 + ((sparkleSeed * (i + 1) * 43) % (CARD_HEIGHT - 60));

                

                const phase = (Date.now() / 250 + i * 2.2) % (Math.PI * 2);

                const size = (Math.sin(phase) + 1) * 2.5 + 0.5;

                

                if (size > 1.2) {

                    ctx.beginPath();

                    ctx.moveTo(px, py - size * 2);

                    ctx.quadraticCurveTo(px, py, px + size * 2, py);

                    ctx.quadraticCurveTo(px, py, px, py + size * 2);

                    ctx.quadraticCurveTo(px, py, px - size * 2, py);

                    ctx.quadraticCurveTo(px, py, px, py - size * 2);

                    ctx.fill();

        function animateLoop(timestamp) {

        }
